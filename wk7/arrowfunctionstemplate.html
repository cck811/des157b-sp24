Text Transformation Functions

Got this simple text transformation functions that convert messages to uppercase and lowercase from wk3-JS-Event-Loop

Named Function

            function mkUpperCase(message) {
                return message.toUpperCase();
            }

            function mkLowerCase(message) {
                return message.toLowerCase();
            }

Function Expression

            const outputMessages = function(message) {
                let output = `Uppercase version: ${mkUpperCase(message)}\n`;
                output += `Lowercase version: ${mkLowerCase(message)}`;
                console.log(output);
            };

Arrow Function

            const outputMessages = message => {
                let output = `Uppercase version: ${mkUpperCase(message)}\n`;
                output += `Lowercase version: ${mkLowerCase(message)}`;
                console.log(output);
            };
